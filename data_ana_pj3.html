get_shap_values <- function(data_subset, label) {
  shap_list <- lapply(1:nrow(data_subset), function(i) {
    shap <- Shapley$new(predictor, x.interest = data_subset[i, feature_names])
    shap_df <- shap$results |> mutate(obs = i, group = label)
    return(shap_df)
  })
  bind_rows(shap_list)
}
